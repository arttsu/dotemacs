* Repeatize keymaps
https://karthinks.com/software/it-bears-repeating/

#+begin_src elisp
  (defun repeatize (keymap)
    "Add `repeat-mode' support to a KEYMAP."
    (map-keymap
     (lambda (_key cmd)
       (when (symbolp cmd)
         (put cmd 'repeat-map keymap)))
     (symbol-value keymap)))
#+end_src
